# ----------------------------------------------------------------------------
# FILOSOFIA DO PROJETO: ZERO-DEPENDENCY
# ----------------------------------------------------------------------------
# Este projeto NAO usa Composer nem NPM. Nao ha vendor/, composer.json,
# package.json ou node_modules. Tudo e PHP nativo + 1 lib JS via CDN.
# Nao adicionar Composer/Slim/Monolog/PHPMailer etc. sem antes discutir —
# isso muda a arquitetura de instalacao (hoje: copiar pasta + apontar o
# document root, sem "composer install").

# ----------------------------------------------------------------------------
# RUNTIME
# ----------------------------------------------------------------------------
php>=8.1
# Confirmado em uso: tipo "mixed" (backend/api.php) -> exige PHP >= 8.0.
# README.md já trava em PHP 8.1+; manter os dois sincronizados se subir a versão.

# ----------------------------------------------------------------------------
# EXTENSOES PHP NATIVAS (vem com o PHP, mas precisam estar habilitadas)
# ----------------------------------------------------------------------------
ext-pdo                 # backend/db.php — acesso a banco via PDO (obrigatória)
ext-openssl             # backend/auth.php, backend/mailer.php — cifra de senha SMTP / JWT
ext-mbstring            # backend/mailer.php — mb_encode_mimeheader (assunto de e-mail com acento)
ext-json                # backend/api.php — json_encode/json_decode (request/response da API)
ext-hash                # uso de hash() para checksums/assinaturas
# password_hash()/password_verify()/random_bytes() são da core do PHP, não exigem extensão extra.

# ----------------------------------------------------------------------------
# DRIVER DE BANCO — instalar SOMENTE o que for usado em produção
# (escolha feita em /setup; backend/db.php verifica extension_loaded() e
# acusa erro amigável se faltar)
# ----------------------------------------------------------------------------
ext-pdo_sqlite          # padrão zero-config (cria db/database.db sozinho)
# ext-pdo_mysql         # se o banco escolhido for MySQL/MariaDB
# ext-pdo_pgsql         # se o banco escolhido for PostgreSQL
# ext-pdo_sqlsrv        # se o banco escolhido for SQL Server (driver da Microsoft)
